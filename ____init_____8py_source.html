127  def read_configuration(self):
128  """!
129  @brief Normalize input configuration based on the configuration
130  definition: split strings into lists, convert to types.
131  """
132  env = {}
133  errors = 0
134  missing = 0
135  keys = list(set(self.REQUIRED_CONFIG + self.OPTIONAL_CONFIG))
136 
137  # Iterate through required and optional config, and read only those variables
138  for key in keys:
139  if key not in self.CONFIG:
140  raise RuntimeError(
141  "Missing configuration option '%s' in adapter CONFIG hash, please fix your code!" % key
142  )
143 
144  # Read default value and enforce requirements
145  if key not in self.env:
146  if key in self.REQUIRED_CONFIG:
147  self.logger.critical(
148  'Missing required environment variable %s (%s)', key, self.CONFIG[key]['help']
149  )
150  missing += 1
151  errors += 1
152  continue
153  if key in self.OPTIONAL_CONFIG:
154  self.logger.debug('Optional environment variable not configured: %s (%s), defaults to "%s"', key, self.CONFIG[key]['help'], self.CONFIG[key]['default'])
155  value = self.CONFIG[key]['default']
156  else:
157  value = self.env[key]
158 
159  # Normalize configuration option
160  option_type = self.CONFIG[key]['type']
161  func_name = 'normalize_config_' + option_type
162  try:
163  func = getattr(self, func_name)
164  except:
165  raise RuntimeError("No normalization function for configuration type '%s' found!" % option_type)
166  try:
167  value = func(value)
168  except Exception as e:
169  self.logger.critical("Invalid value '%s' for configuration '%s': %s", value, key, e)
170  errors += 1
171  continue
172 
173  # Write normalized value into configuration hash
174  env[key] = value
175 
176  # Terminate if invalid configuration
177  if missing > 0:
178  raise eva.exceptions.MissingConfigurationException(
179  'Missing %d required configuration values' % missing
180  )
181  if errors > 0:
182  raise eva.exceptions.InvalidConfigurationException(
183  'Encountered %d errors while reading configuration' % errors
184  )
185 
186  # Drop non-normalized values
187  self.env = env

199 def retry_n(func, args=(), kwargs={}, interval=5, exceptions=(Exception,), warning=1, error=3, give_up=5, logger=logging):
200  """
201  Call 'func' and, if it throws anything listed in 'exceptions', catch it and retry again
202  up to 'give_up' times. If give_up is <= 0, retry indefinitely.
203  Checks that error > warning > 0, and give_up > error or give_up <= 0.
204  """
205  assert (warning > 0) and (error > warning) and (give_up <= 0 or give_up > error)
206  tries = 0
207  while True:
208  try:
209  return func(*args, **kwargs)
210  except exceptions as e:
211  tries += 1
212  if give_up > 0 and tries >= give_up:
213  logger.error('Action failed %d times, giving up: %s' % (give_up, e))
214  return False
215  if tries >= error:
216  logfunc = logger.error
217  elif tries >= warning:
218  logfunc = logger.warning
219  else:
220  logfunc = logger.info
221  logfunc('Action failed, retrying in %d seconds: %s' % (interval, e))
222  time.sleep(interval)
