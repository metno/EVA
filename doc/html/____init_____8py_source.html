94  def read_configuration(self):
95  """!
96  @brief Normalize input configuration based on the configuration
97  definition: split strings into lists, convert to types.
98  """
99  env = {}
100  errors = 0
101  missing = 0
102  keys = list(set(self.REQUIRED_CONFIG + self.OPTIONAL_CONFIG))
103 
104  # Iterate through required and optional config, and read only those variables
105  for key in keys:
106  if key not in self.CONFIG:
107  raise RuntimeError(
108  "Missing configuration option '%s' in adapter CONFIG hash, please fix your code!" % key
109  )
110 
111  # Read default value and enforce requirements
112  if key not in self.env:
113  if key in self.REQUIRED_CONFIG:
114  self.logger.critical(
115  'Missing required environment variable %s (%s)', key, self.CONFIG[key]['help']
116  )
117  missing += 1
118  errors += 1
119  continue
120  if key in self.OPTIONAL_CONFIG:
121  self.logger.debug('Optional environment variable not configured: %s (%s), defaults to "%s"', key, self.CONFIG[key]['help'], self.CONFIG[key]['default'])
122  value = self.CONFIG[key]['default']
123  else:
124  value = self.env[key]
125 
126  # Normalize configuration option
127  option_type = self.CONFIG[key]['type']
128  func_name = 'normalize_config_' + option_type
129  try:
130  func = getattr(self, func_name)
131  except:
132  raise RuntimeError("No normalization function for configuration type '%s' found!" % option_type)
133  try:
134  value = func(value)
135  except Exception as e:
136  self.logger.critical("Invalid value '%s' for configuration '%s': %s", value, key, e)
137  errors += 1
138  continue
139 
140  # Write normalized value into configuration hash
141  env[key] = value
142 
143  # Terminate if invalid configuration
144  if missing > 0:
145  raise eva.exceptions.MissingConfigurationException(
146  'Missing %d required configuration values' % missing
147  )
148  if errors > 0:
149  raise eva.exceptions.InvalidConfigurationException(
150  'Encountered %d errors while reading configuration' % errors
151  )

158  """
159  Call 'func' and, if it throws anything listed in 'exceptions', catch it and retry again

161  Checks that error > warning > 0, and give_up > error or give_up <= 0.

165  while True:
166  try:
167  return func(*args, **kwargs)
168  except exceptions as e:
169  tries += 1
170  if give_up > 0 and tries >= give_up:
171  logger.error('Action failed %d times, giving up: %s' % (give_up, e))
172  return False
173  if tries >= error:
174  logfunc = logger.error
175  elif tries >= warning:
176  logfunc = logger.warning
177  else:
178  logfunc = logger.info
